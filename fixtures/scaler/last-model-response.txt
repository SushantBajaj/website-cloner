            if (entry.isIntersecting) {
                stickyNav.style.display = 'none';
            } else {
                stickyNav.style.display = 'block';
            }
        },
        {
            rootMargin: `-${navbarHeight}px 0px 0px 0px`, // Trigger when hero is out of view
            threshold: 0 // Trigger as soon as it leaves
        }
    );

    if (heroSection) {
        observer.observe(heroSection);
    }

    // Program tabs functionality
    const programTabs = document.querySelectorAll('.program-tab');
    const programNavItems = document.querySelectorAll('.program-nav-item');
    const programDetailsCards = document.querySelectorAll('.program-card'); // Assuming multiple cards for different programs

    function activateProgram(programId) {
        programDetailsCards.forEach(card => {
            card.style.display = 'none';
        });
        const activeCard = document.getElementById(`${programId}-details`);
        if (activeCard) {
            activeCard.style.display = 'block';
        }
    }

    programTabs.forEach(tab => {
        tab.addEventListener('click', () => {
            programTabs.forEach(t => t.classList.remove('active'));
            tab.classList.add('active');
            // In a real scenario, this would filter/load programs based on tab
            // For this static page, we'll just activate the first program nav item
            programNavItems[0].click(); // Simulate click on first program nav item
        });
    });

    programNavItems.forEach(item => {
        item.addEventListener('click', () => {
            programNavItems.forEach(i => i.classList.remove('active'));
            item.classList.add('active');
            const programId = item.dataset.program;
            activateProgram(programId);
        });
    });

    // Initialize first program tab and item as active
    if (programTabs.length > 0) {
        programTabs[0].classList.add('active');
    }
    if (programNavItems.length > 0) {
        programNavItems[0].classList.add('active');
        activateProgram(programNavItems[0].dataset.program);
    }
});
